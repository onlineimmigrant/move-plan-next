// Added event listener
useEffect(() => {
  const handleAddProductEvent = () => {
    handleAdd();
  };
  
  window.addEventListener('addProduct', handleAddProductEvent);
  
  return () => {
    window.removeEventListener('addProduct', handleAddProductEvent);
  };
}, [handleAdd]);
